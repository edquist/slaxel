#/bin/sh

# This file is sourced by blahp before submitting the job to pbs
# Anything printed to stdout is included in the submit file.
# For example, to set a default walltime of 24 hours in PBS, you
# could uncomment this line:

# echo "#PBS -l walltime=24:00:00"

# blahp allows arbitrary attributes to be passed to this script on a per-job
# basis.  If you add the following to your HTCondor-G submit file:

#+remote_cerequirements = NumJobs == 100 && foo = 5

# Then an environment variable, NumJobs, will be exported prior to calling this
# script and set to a value of 100.  The variable foo will be set to 5.

# You could allow users to set the walltime for the job with the following
# customization (PBS syntax given; adjust for the appropriate batch system):
#echo "#PBS -l pmem=3950m" klb, 4/14/2015

if [ -z "$maxWallTime" ]; then
    maxWallTime="48:00:00"
fi
if [ -z "$maxMemory" ]; then
    maxMemory="4000"
fi

# Set job priorities. Higher is less prioritized
SUB_USER=$(whoami)
if [ "$SUB_USER" == "cmspilot" ]; then
    PRIORITY=650
elif [ "$SUB_USER" == "cmslocal" ]; then
    PRIORITY=350
else
    PRIORITY=100
fi

if [[ "$blah_opt_mpinodes" == "1" ]]; then
    PRIORITY=$(($PRIORITY + 400));
elif [[ "$blah_opt_mpinodes" == "4" ]]; then
    PRIORITY=$(($PRIORITY + 200));
fi

if [ "$SUB_USER" == "lcgadmin" ]; then
    PRIORITY=0
fi

echo "# Begin  /etc/blahp/slurm_local_submit_attributes.sh"

echo "#SBATCH --mem=$maxMemory"
echo "#SBATCH --time=$maxWallTime"
PRIO="default"
echo "$(date) in  Mem: $maxMemory Cores: $blah_opt_mpinodes Wall: $maxWallTime User: $SUB_USER Prio: $PRIORITY QOS: ${ACCREQOS:-unknown} QUEUE: $PRIO GPU: ${RequestGPUs:-unknown} Whole: ${WholeNode:-unknown} " >> /tmp/melo-ce2-debug-meloam.txt 2>/dev/null

if [ "$SUB_USER" == "lcgadmin" ]; then
    #echo "#SBATCH --reservation=cmstest"
    echo "#SBATCH --partition=sam --account=cmsadmin"
    PRIO="cmstest"
elif [[ "${RequestGPUs:-0}" -gt 0 ]]; then
    echo "#SBATCH --account=cms_gpu_acc"
    echo "#SBATCH --partition=pascal"
    echo "#SBATCH --gres=gpu:$RequestGPUs"
#    echo 'eval /usr/local/bin/config_pkg -sh -a cuda8.0'
    PRIO="gpu"
# The user mapping is also enforced by the job routes
# Do we want whole nodes? Not sure where the quotes come from... 
elif [[  "${WholeNode}" == "\"true\"" ||  "${WholeNode}" == "\"True\"" ]]; then
    # echo "#SBATCH --partition=nogpfs,production"
    echo "#SBATCH --partition=nogpfs"
    echo "#SBATCH --mem=0"
    echo "#SBATCH --exclusive"
    # Really prioritize whole node jobs
    PRIORITY=0
    PRIO="whole"
elif [[ ($SUB_USER == "cmspilot" || $SUB_USER == "cmslocal") ]]; then
    # Send pilot jobs to both partitions

    REGPRIO_COUNT=$(squeue -A cms -p production -t pending -o %C 2>/dev/null | grep $blah_opt_mpinodes | wc -l)
    LOWPRIO_COUNT=$(squeue -A cms_lowprio -p production -t pending -o %C 2>/dev/null | grep $blah_opt_mpinodes | wc -l)

    # Let some pilot jobs be preemptable
    if [[ "$LOWPRIO_COUNT" -lt "$REGPRIO_COUNT" ]]; then
        echo "#SBATCH --account=cms_lowprio"
        PRIO="low"
    else
        echo "#SBATCH --partition=production"
    fi
fi



if [[ "$PRIORITY" != "0" ]]; then
    echo "#SBATCH --priority=$PRIORITY"
    echo "#SBATCH --nice=${PRIORITY}"
fi

echo 'export PATH=$PATH:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin'
echo 'export X509_CERT_DIR=/cvmfs/oasis.opensciencegrid.org/osg-software/osg-wn-client/certificates'
echo "# End    /etc/blahp/slurm_local_submit_attributes.sh"

echo "$(date) out Mem: $maxMemory Cores: $blah_opt_mpinodes Wall: $maxWallTime User: $SUB_USER Prio: $PRIORITY QOS: ${ACCREQOS:-unknown} QUEUE: $PRIO GPU: ${RequestGPUs:-unknown} PWD: ${PWD}" >> /tmp/melo-ce2-debug-meloam.txt 2>/dev/null
